COMANDO PARA WORKBENCH

create database [nome do banco]; cria um banco de dados

create table [nome da tabela]; cria uma tabela

use [nome do banco]; exibe as info do banco

describe ou desc [nome da tabela]; exibe os campos da tabelas escolhida

update [TABELA] set [coluna: CAMPO='NOME'] WHERE [linha:ID_CLIENTE='1'];

selec * from [TABELA]; exibe todos os dados da tabela
